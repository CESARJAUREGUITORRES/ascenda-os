var sh = _sh("NOTAS_PACIENTES"); var lr = sh.getLastRow(); if (lr < 2) throw new Error("Sin notas registradas."); // Leer las 9 columnas de la hoja var data = sh.getRange(2, 1, lr - 1, 9).getValues(); for (var i = 0; i < data.length; i++) { if (_norm(data[i][0]) !== notaId) continue; // Verificar permisos var usuarioNota = _norm(data[i][6]); var esAdmin = s.rol === "ADMIN" || s.rol === "ADMINISTRADOR"; var esPropietario= _norm(s.asesor) === usuarioNota || _norm(s.usuario) === usuarioNota; if (!esAdmin && !esPropietario) throw new Error("Solo puedes editar tus propias notas."); // Actualizar TEXTO (col 5 base-1) y marcar como editado en TIPO_NOTA (col 8) sh.getRange(i + 2, 5).setValue(nuevoTexto); var tipoActual = _norm(data[i][7]); if (tipoActual && tipoActual.indexOf("(editado)") < 0) { sh.getRange(i + 2, 8).setValue(tipoActual + " (editado)"); } Logger.log("api_updateNotaPaciente360 OK — id: " + notaId + " | por: " + s.asesor); return { ok: true, msg: "Nota actualizada." }; } throw new Error("Nota no encontrada: " + notaId); } function api_updateNotaPaciente360T(token, notaId, nuevoTexto) { _setToken(token); return api_updateNotaPaciente360(notaId, nuevoTexto); } // ===== CTRL+F: api_updateNotaPaciente360_END ===== // ═══════════════════════════════════════════════════ // PATCH-GS24-02 · ELIMINAR NOTA // ===== CTRL+F: api_deleteNotaPaciente360 ===== // ═══════════════════════════════════════════════════ /** * api_deleteNotaPaciente360 * Elimina una nota de NOTAS_PACIENTES por su ID. * La fila se borra físicamente del Sheet. * * Acceso: el mismo usuario que la creó, o ADMIN/ADMINISTRADOR. * * @param {string} notaId ID único de la nota (col A de la hoja) */ function api_deleteNotaPaciente360(notaId) { var s = cc_requireSession(); for (var i = 0; i < data.length; i++) { if (_norm(data[i][0]) !== notaId) continue; // Verificar permisos var usuarioNota = _norm(data[i][6]); var esAdmin = s.rol === "ADMIN" || s.rol === "ADMINISTRADOR"; Logger.log("api_deleteNotaPaciente360 OK — id: " + notaId + " | por: " + s.asesor); return { ok: true, msg: "Nota eliminada." }; } throw new Error("Nota no encontrada: " + notaId); } function api_deleteNotaPaciente360T(token, notaId) { _setToken(token);
